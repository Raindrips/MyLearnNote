class drawArc{
    // 弧
    void drawArc();
    // 弦
    void drawChord();

    void drawConvexPolygon();

    void drawEllipse();

    void drawLine()
        // 路径； 想到哪就去哪，但是必须是直的
        void drawLines(); // 多条线drawPath()
    // 按QPainter指令绘制 区分drawImage、drawPixmap，大家可以试一把
    void  drawPicture();
    // 扇形
    void  drawPie();

    void drawPixmap();
    // 多个点
    void drawPoints();
    //多边形
    void drawPolygon();

    //多折线drawRect() 矩 形 drawRects() 多个矩形
    void drawPolyline();
    //圆角矩形
    drawRoundRect();
    // 文字
    drawText();
    // 平铺图像
    drawTiledPixmap();
    //绘制折线
    drawLineSegments();

};
